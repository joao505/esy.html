// Definindo a classe Task para representar uma tarefa
class Task {
  constructor(description, priority) {
    this.description = description;
    this.priority = priority;
  }
}

// Função para adicionar uma nova tarefa à lista
function adicionarTarefa(descricao, prioridade) {
  let novaTarefa = new Task(descricao, prioridade);
  listaDeTarefas.push(novaTarefa);
  console.log("Tarefa adicionada: " + descricao);
}

// Função para remover uma tarefa da lista
function removerTarefa(descricao) {
  for (let i = 0; i < listaDeTarefas.length; i++) {
    if (listaDeTarefas[i].description === descricao) {
      listaDeTarefas.splice(i, 1);
      console.log("Tarefa removida: " + descricao);
      return;
    }
  }
  console.log("Tarefa não encontrada: " + descricao);
}

// Função para exibir todas as tarefas na lista
function listarTarefas() {
  console.log("Lista de Tarefas:");
  listaDeTarefas.forEach((tarefa, index) => {
    console.log(`${index + 1}. ${tarefa.description} - Prioridade: ${tarefa.priority}`);
  });
}

// Array para armazenar as tarefas
let listaDeTarefas = [];

// Adicionando algumas tarefas de exemplo
adicionarTarefa("Lavar o carro", "Alta");
adicionarTarefa("Fazer compras", "Média");
adicionarTarefa("Estudar para o teste", "Alta");
adicionarTarefa("Passear com o cachorro", "Baixa");

// Listando todas as tarefas
listarTarefas();

// Removendo uma tarefa
removerTarefa("Fazer compras");

// Listando as tarefas atualizadas
listarTarefas();
